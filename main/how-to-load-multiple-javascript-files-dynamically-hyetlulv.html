How to load multiple JavaScript files dynamically
Create a <script> element for each file that you want to load dynamically.
let script1 = document.createElement('script');
let script2 = document.createElement('script');
let script3 = document.createElement('script');
Set the src attribute to the path of the JavaScript file.
script1.src = 'file1.js';
script2.src = 'file2.js';
script3.src = 'file3.js';
Append the <script> element to the document's <head> tag.
document.head.appendChild(script1);
document.head.appendChild(script2);
document.head.appendChild(script3);

Repeat the process for any other JavaScript files you need.
Create a callback function to execute when all JavaScript files have loaded.
let onLoaded = function() {
  
}
Add an onload attribute to the last <script> element and set it to the callback function.
script3.onload = onLoaded;
Execute the callback function when all JavaScript files have loaded. Example:
<script>
 let script1 = document.createElement('script');
 let script2 = document.createElement('script');
 let script3 = document.createElement('script');
 script1.src = 'file1.js';
 script2.src = 'file2.js';
 script3.src = 'file3.js';
 document.head.appendChild(script1);
 document.head.appendChild(script2);
 document.head.appendChild(script3);
 let onLoaded = function() {
  // Do something when all files have loaded
 }
 script3.onload = onLoaded;
</script>
